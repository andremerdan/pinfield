## Install

```bash
npm install --save pinfield
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'pinfield'
import 'pinfield/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
